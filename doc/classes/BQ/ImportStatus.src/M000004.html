    def initialize(table,import_id)
      # Encode destination table name
      url = "https://www.googleapis.com/bigquery/v1/tables/#{CGI.escape(table)}/imports/#{import_id}"

      # Send data import request
      result = BQ.request(:post, {}, url)

      return result
    end
