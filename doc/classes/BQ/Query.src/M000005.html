    def execute(sql)
      # Add a ";" if it's missing
      sql += ";" if not sql.end_with?(';')

      # Use a remote procedure call (RPC) to query
      data = {:params => {:q => sql}, :method => 'bigquery.query'}

      # Send a query request
      result = BQ.request(:post, data.to_json)

      return result
    end
